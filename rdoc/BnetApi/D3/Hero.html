def get
  BnetApi.api = :d3
  json = BnetApi::Api.get("profile/#{battle_tag}/hero/#{id}")
  
  @id = json["id"]
  @class_name = json["class"]
  @dead = json["dead"]
  @gender = json["gender"] == 0 ? "Male" : "Female"
  @hardcore = json["hardcore"]
  @last_updated = Time.at(json["last-updated"])
  @level = json["level"]
  @name = json["name"]
  @paragon_level = json["paragonLevel"]
  
  self
end
